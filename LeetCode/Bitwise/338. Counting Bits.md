# 338. Counting Bits

# Solution
```
class Solution {
public:
    vector<int> countBits(int n) {
        vector<int> bin(n + 1);
        for(int i = 0; i <= n; ++i){
            int count = __builtin_popcount(i);
            bin[i] = (int)count;
        }
        return bin;
    }
};
```
